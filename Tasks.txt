# Attendance
CREATE TABLE IF NOT EXISTS attendance (id INTEGER PRIMARY KEY, idPaymentMethod INTEGER NOT NULL, patient TEXT NOT NULL, installments INTEGER NOT NULL, attendanceDate LONG NOT NULL, fullValue REAL NOT NULL, receiveValue REAL NOT NULL, machineTaxValue REAL NOT NULL, clinicValue REAL NOT NULL, paid INTEGER, obs TEXT, FOREIGN KEY(idPaymentMethod) REFERENCES paymentMethod(id))


CREATE TABLE IF NOT EXISTS installment (id INTEGER PRIMARY KEY, idAttendance INTEGER NOT NULL, number INTEGER NOT NULL, value REAL NOT NULL, paid INTEGER, expectedPaymentDate LONG, paymentDate LONG, FOREIGN KEY(idAttendance) REFERENCES attendance(id))
